# Set up the figure
plt.figure(figsize=(10, 6))

# Loop through each row and plot each data point using the index as the x-axis
for index, row in most_popular.iterrows():
    plt.scatter(index, row['bottles_sold'], label=f"Item {row['item_number']}")
    
    # Annotate item_number if bottles_sold > 250, no decimals
    if row['bottles_sold'] > 220:
        plt.annotate(
            f'{int(row["item_number"])}',  # Convert item_number to integer to remove decimals
            (index, row['bottles_sold']),
            textcoords="offset points", 
            xytext=(0, 5), 
            ha="center"
        )

# Add labels and title
plt.xlabel('Zipcode')
plt.ylabel('Bottles Sold')
plt.title('Bottles Sold in each Zipcode')
